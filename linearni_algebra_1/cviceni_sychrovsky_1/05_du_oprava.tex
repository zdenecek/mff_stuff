\begin{equation*}
A =
\begin{pmatrix}
3 & 2 \\
2 & 4
\end{pmatrix} 
\end{equation*}
\begin{equation*}
A^2 =
\begin{pmatrix}
6 & 0 \\
0 & 6
\end{pmatrix} 
\end{equation*}
\begin{equation*}
A^4 = A^2 \cdot A^2  = 6 \cdot A^2 =
\begin{pmatrix}
1 & 0 \\
0 & 1
\end{pmatrix}
= I_2
\end{equation*}
\begin{equation*}
A^{5!} = A^{120}  = (A^4)^{30} = (I_2)^{30} = I_2 = 
\begin{pmatrix}
1 & 0 \\
0 & 1
\end{pmatrix} 
\end{equation*}
